Private Sub CommandButton2_Click()
    Dim lb As MSForms.ListBox
    Dim Item As Variant
    Dim OutputFilePath As String
    Dim OutputFile As Integer
    Dim i As Long

    ' リストボックスの参照を取得
    Set lb = UserForm1.ListBox1 ' リストボックスの名前を適宜変更
    
    ' 出力ファイルのパスを指定
    OutputFilePath = ThisWorkbook.Path & "\a.txt" ' 出力先のパスを適宜変更
    
    ' テキストファイルを新規作成または上書き
    OutputFile = FreeFile
    Open OutputFilePath For Output As OutputFile
    
    For i = 0 To lb.ListCount - 1
    If Not lb.Selected(i) Then
        Print #OutputFile, lb.List(i)
    End If
    Next
        
    ' テキストファイルを閉じる
    Close OutputFile
  
End Sub
